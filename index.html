Getting Started

Install Tyndere using npm or include it directly in your HTML.

npm install tyndere

Or include it via CDN:

<script src="https://unpkg.com/tyndere@latest/dist/tyndere.min.js"></script>

Examples

Creating a Component

// Define a simple counter component
const Counter = Tyndere.component({
  state: { count: 0 },
  render: function(state, props) {
    return Tyndere.h('div', { class: 'counter' }, [
      Tyndere.h('p', {}, `Count: ${state.count}`),
      Tyndere.h('button', { 
        onclick: () => this.setState({ count: state.count + 1 }) 
      }, 'Increment')
    ]);
  }
});

Using HTML Templates

// Tyndere works with HTML templates too
const TodoApp = Tyndere.component({
  state: { todos: [], text: '' },
  render: function(state) {
    return `
      <div class="todo-app">
        <h3>TODO</h3>
        <ul>
          ${state.todos.map(todo => `<li>${todo}</li>`).join('')}
        </ul>
        <input 
          value="${state.text}" 
          onInput="${Tyndere.bind(this, 'handleInput')}"
        />
        <button onClick="${Tyndere.bind(this, 'addTodo')}">
          Add #${state.todos.length + 1}
        </button>
      </div>
    `;
  },
  handleInput: function(e) {
    this.setState({ text: e.target.value });
  },
  addTodo: function() {
    this.setState(state => ({
      todos: [...state.todos, state.text],
      text: ''
    }));
  }
});

CSS Integration

// Tyndere has built-in CSS support
const StyledButton = Tyndere.component({
  styles: `
    .styled-button {
      background: var(--primary-color);
      color: white;
      border: none;
      padding: 8px 16px;
      border-radius: 4px;
      cursor: pointer;
      transition: opacity 0.2s;
    }
    .styled-button:hover {
      opacity: 0.9;
    }
  `,
  render: function(_, props) {
    return Tyndere.h('button', { 
      class: 'styled-button',
      onclick: props.onClick
    }, props.children);
  }
});
